CMAKE_DEPENDENT_OPTION(ENABLE_NGW "Build NextGIS Web API format" ON "CURL_FOUND" OFF)
if(ENABLE_NGW)
    set(TARGET_NAME com_ngw)

    add_definitions(-DNGW_ENABLED)
	set(FORMAT_DEFINITIONS ${FORMAT_DEFINITIONS} "-DNGW_ENABLED" PARENT_SCOPE)

    include_directories(${CMAKE_CURRENT_SOURCE_DIR})

    set(LIB_HHEADERS
    )

    set(LIB_HHEADERS_PRIV
        ogr_ngw.h
    )

    set(LIB_CSOURCES
        ngw_api.cpp
        gdalngwdataset.cpp
        ogrngwdriver.cpp
        ogrngwlayer.cpp
    )

    add_library(${TARGET_NAME} OBJECT ${LIB_HHEADERS} ${LIB_HHEADERS_PRIV} ${LIB_CSOURCES})

    # build target
    set(TARGET_OBJECTS ${TARGET_OBJECTS} $<TARGET_OBJECTS:${TARGET_NAME}> PARENT_SCOPE)
else()
    warning_msg("Skip ngw")
endif()
